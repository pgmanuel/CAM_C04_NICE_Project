# -----------------------------------------------------------------------------
# 12. GRAPH PROCESSING & SNOMED POLYHIERARCHY
# networkx is required for modelling the SNOMED CT concept hierarchy as a
#   directed acyclic graph (DAG). It provides graph traversal (finding all
#   parent/child concepts), centrality metrics (degree, betweenness), and
#   community detection (identifying multimorbidity clusters). Without this,
#   the Hierarchy Explorer tool cannot navigate the SNOMED polyhierarchy,
#   meaning the agent will miss bridging concepts in comorbidity code lists.
# python-louvain provides the Louvain community detection algorithm, which
#   identifies natural clusters in the multimorbidity co-occurrence graph
#   and is referenced in the academic research for this project.
# ruptures provides change-point detection for time series — used in the
#   semantic drift monitoring pipeline to automatically flag SNOMED codes
#   whose usage has undergone a structural break (likely deprecation).
# -----------------------------------------------------------------------------
networkx>=3.1
python-louvain>=0.16
ruptures>=1.1.9


# -----------------------------------------------------------------------------
# 13. PROVENANCE, AUDIT TRAILS & METADATA (PROV-O)
# rdflib is the Python library for working with RDF (Resource Description
#   Framework) — the data model used by the W3C PROV-O provenance ontology.
#   PROV-O is the formal standard for representing the lineage of a digital
#   artefact, and the NICE research document specifically recommends it for
#   building auditable, interoperable code list provenance records. rdflib
#   allows you to serialise provenance records as JSON-LD or Turtle format,
#   which can be archived alongside each code list as machine-readable evidence.
# jsonschema validates that provenance records conform to the expected schema
#   before they are saved — preventing silent structural errors in the audit trail.
# -----------------------------------------------------------------------------
rdflib>=7.0.0
jsonschema>=4.22.0


# -----------------------------------------------------------------------------

#   is compatible with the same SHAP TreeExplainer interface.
# chardet detects the byte-level encoding of files before reading them.
#   This is non-optional for NHS clinical data: files exported from legacy
#   GP clinical systems are frequently Latin-1 (ISO-8859-1) rather than
#   UTF-8, and reading them incorrectly silently corrupts code descriptions.
#   Used in scripts/initialize_gold_standard.py and src/ingestion/*.
# chardet is referenced in REPO-01_data_ingestion_scripts.md.

chardet>=5.2.0
